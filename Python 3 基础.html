def single(cls):
    cls_obj = {}  # 一个类对应一个实例
    def f():
        if cls not in cls_obj:
            obj = cls()  # 创建实例
            cls_obj[cls] = obj
        return cls_obj[cls]
    return f

@single
class Message:
    def __init__(self):
        return

print(m1)
