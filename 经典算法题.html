# 全排列

问题：求序列a,b,c的全排列。

a = ['a', 'b', 'c']
res = []

def f(A):
    if len(res) == 2:
        print(res)
    else:
        for i in a:
            if i not in res:
                res.append(i)
                f(A)
                res.remove(i)
f(a)
